邮件发送配置
说明：
  • 注意，邮箱需要开启smtp功能。
详细：
//邮件发送配置
$config['SMTP_HOST']='smtp.qq.com';//smtp服务器地址
$config['SMTP_PORT']=25;//smtp服务器端口
$config['SMTP_SSL']=false;//是否启用SSL安全连接，gmail需要启用sll安全连接
$config['SMTP_USERNAME']='404352772@qq.com';//smtp服务器帐号，如：你的qq邮箱
$config['SMTP_PASSWORD']='123456';//smtp服务器帐号密码，如你的qq邮箱密码
$config['SMTP_AUTH']=true;//启用SMTP验证功能，一般需要开启
$config['SMTP_CHARSET']='utf-8';//发送的邮件内容编码
$config['SMTP_FROM_TO']='404352772@qq.com';//发件人邮件地址
$config['SMTP_FROM_NAME']='CanPHP官方';//发件人姓名
$config['SMTP_DEBUG']=false;//是否显示调试信息 
//邮件发送配置结束
